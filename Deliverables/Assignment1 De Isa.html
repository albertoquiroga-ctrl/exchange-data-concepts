data1 = [ #imports image data
    ['A', 2 ],
    ['B', 7 ],
    ['C', 9 ]
]

total = 10

df1 = pd.DataFrame(data1, columns=['Group', '+count']) #creates data frame

df1['-count'] = total - df1['+count'] #adds a column called -count that contains the amount of -s in the image

df1['total'] =  total #adds a column called total that has all of the observations per group

df1['+prop'] = df1['+count']/df1['total'] #calculates proportion of +s over total

df1['-prop'] = df1['-count']/df1['total'] #calculates proportion of -s over total

df1['entropy'] = - df1['+prop'] * np.log2(df1['+prop']) - df1['-prop'] * np.log2(df1['-prop']) #calculates entropy according to formula

df1['entropy'] = np.round(df1['entropy'], 2) #rounds entropy to 2 decimal places

print(df1)
